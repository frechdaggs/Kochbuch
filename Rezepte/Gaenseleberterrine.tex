\begin{MyRecipe}{Gänseleberterrine}{ca. \Calc{5}{\x} Terrinen}{ca. einen Tag}
	\ingredient[\Calc{0.5}{\x}]{kg}{Gänseleber}
	\ingredient[\Calc{0.5}{\x}]{kg}{mag. Schweinebauch}
	\ingredient[\Calc{0.2}{\x}]{kg}{Kalbsleber}
	\ingredient[\Calc{0.2}{\x}]{kg}{Schweinenüsschen}
	\ingredient[\Calc{1}{\x}]{kl.}{Zwiebel}
	\ingredient[\Calc{2}{\x}]{}{Zwiebeln}
	\ingredient[\Calc{1}{\x}]{}{Knoblauchzehen}
	\ingredient[\Calc{0.2}{\x}]{kg}{frischer Speck}
	\ingredient[\Calc{3}{\x}]{}{Eigelb}
	\ingredient[\Calc{2}{\x}]{}{Schalotten}
	\ingredient[\Calc{6}{\x}]{Bl.}{Gelatine}
	\ingredient[]{}{Lorbeerblatt}
	\ingredient[]{}{Pfefferkörner}
	\ingredient[]{}{Thymian}
	\ingredient[]{}{Majoran}
	\ingredient[]{}{Piment}
	\ingredient[]{}{Gänsefett}
	\ingredient[]{}{Weißwein}
	\ingredient[]{}{Noilly Prat}
	\ingredient[]{}{Portwein}
	\ingredient[]{}{Cognac}
		
		
	\textbf{Hinweis:}\\
	Sollen die Pasteten rechtzeitig fertig sein, muss spätestens Vormittag der Vortages begonnen werden.\par\bigskip

	\textbf{Vorbereitung Morgens (ca. 2~h):}\\
	Schweinebauch, \Calc{1}{\x}~kl.~Zwiebel, \Calc{2}{\x}~Pimentkörnern, \Calc{1}{\x}~Lorbeerblatt und \Calc{1}{\x}~TL~Pfefferkörnern mit kaltem Wasser abdecken und ca. 2~h weich kochen.\par
	Brühe durchseihen, das Fleisch in Fleischwolf-gerechte Stücke schneiden.\par\bigskip
	
	\textbf{Stehen lassen (ca. 3~h):}\\	
	Fleisch und Brühe gut abkühlen lassen, sodass Fett abgeschöpft werden kann.\par
	Ein Drittel der Gänseleber in Cognac einlegen.\par\bigskip
	
	\textbf{Zubereitung Nachmittags (ca. 1.5~h):}\\
	\textit{Farce:}\\	
	Frischen Speck würfeln und in Gefrierschrank legen.\par
	Restl. Gänseleber, Kalbsleber und Nüsschen grob würfeln. \Calc{2}{\x}~Zwiebeln und \Calc{1}{\x}~Knoblauchzehen fein würfeln. Zutaten durch den Fleischwolf lassen. Zusammen mit dem Schweinebauch durch den Fleischwolf lassen. Zuletzt den angefrorenen frischen Speck durch lassen. Danach alles zusammen ein zweites mal durch den Fleischwolf drehen.\par
	Etwas Weißwein, etwas Noilly Prat, \Calc{3}{\x}~Eigelb, kräftig Salz und Pfeffer vermischen und rasch, aber gründlich mit Farce verrühren.\par\medskip
		
	\textit{Jus:}\\	
	\Calc{2}{\x}~Schalotten fein würfeln. Mit etwas Brühe, Weißwein, Noilly Prat und Protwein zu Jus reduzieren, durchseihen und abkühlen lassen.\par\medskip
	
	\textit{Terrinen füllen}:\\
	Maximal zu etwa $\frac34$ füllen: Farce -- eingelegte Gänseleber -- Jus -- Farce\par\bigskip
	
	\textbf{Backen (ca. 1.5~h):}\\
	In Wasserbad bei \SI{175}{\celsius} Umluft, ca. 80'' backen.\par\bigskip
	
	\textbf{Aspik (ca. 20''):}\\
	Ausgetretenen Saft abgießen und mit Brühe zu \Calc{0.5}{1}~Liter auffüllen. Mit Gelatine nach Anleitung zu Aspik verarbeiten und wieder über die Terrinen gießen.\\
	Über Nacht kühl stellen.
	\par\bigskip

	
	
				
	\end{MyRecipe}
