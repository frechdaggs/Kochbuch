\begin{MyRecipe}{Palak Paneer}{\Calc{4}{\x} Personen}{45''}
	\ingredient[\Calc{0.3}{\x}]{kg} {Paneer}
	\ingredient[\Calc{0.2}{\x}]{kg} {Zwiebel(n)}
	\ingredient[\Calc{2}{\x}]{Zehe} {Knoblauch}
	\ingredient[\Calc{0.4}{\x}]{kg} {gehackte Tomaten}
	\ingredient[\Calc{0.4}{\x}]{kg} {Spinat, fein gehackt}
	\ingredient[\Calc{50}{\x}]{ml} {Sahne}
	\ingredient[\Calc{2}{\x}]{TL} {Kreuzkümmel}
	\ingredient[\Calc{1}{\x}]{TL} {Kurkuma}
	\ingredient[\Calc{0.5}{\x}]{TL} {Koriander-samen}
	\ingredient[\Calc{0.5}{\x}]{TL} {Muskatblüte}
	\ingredient[\Calc{0.5}{\x}]{TL} {Kardamom (schwarz)}
	\ingredient[\Calc{1}{\x}]{} {Lorbeer}
	\ingredient[\Calc{0.5}{\x}]{TL} {Nelken}
	\ingredient[\Calc{0.5}{\x}]{TL} {Zimt}
	\ingredient[\Calc{0.5}{\x}]{TL} {Pfeffer}
	\ingredient[\Calc{1}{\x}]{TL} {Garam Masala}
	\ingredient[\Calc{2}{\x}]{cm} {Ingwer}
	\ingredient[]{} {Chili}
	\ingredient[]{} {Salz}
	\ingredient[]{}{Pflanzenöl}
	
	
		
	\textbf{Gewürzmischung:}\\
	Von Kreuzkümmel bis Zimt im Topf kurz anrösten bis es anfängt zu duften, dann zu Pulver zerstoßen/mahlen.\\\par\bigskip
	
	\textbf{Palak:}\\
	Zwiebeln, Knoblauch und Ingwer freien hacken. Öl und Butter in einem Topf oder im Wok erhitzen und Gewürzmischung angehen lassen, Zwiebeln, Knoblauch und Ingwer zugeben und goldbraun braten. Tomaten zufügen, umrühren und zugedeckt ca. 5 min. kochen. Mit Salz und Chili würzen. Spinat und Sahne zugeben. Zugedeckt bei geringer Hitze ca. 10 min. köcheln lassen, dann anpürieren. Zum Schluss Paneer zugeben und einige Minuten weiter köcheln lassen.\\\par\bigskip
	
	\textbf{Dazu passt:}\\
	Kurkuma-Reis (mit Sultaninen oder Korinthen).

	
	
				
	\end{MyRecipe}
